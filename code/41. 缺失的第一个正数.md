# [41. 缺失的第一个正数](https://leetcode-cn.com/problems/first-missing-positive/)

+ 注意事项：算法的时间复杂度应为O(n)，并且只能使用常数级别的额外空间。

```cpp
class Solution {
public:
    int firstMissingPositive(vector<int>& nums) {
        unordered_set<int> s;
        for(auto &it:nums)
            if(it>0) s.insert(it);
        int n=0;
        while(s.find(++n)!=s.end()){}
        return n;
    }
};
```

改装Hash

```cpp
class Solution {
public:
    int firstMissingPositive(vector<int>& nums) {
        int n=nums.size();
        for(auto &it:nums)
            if(it<=0) it=n+1;
        for(int i=0;i<n;++i){
            int tmp=abs(nums[i]);
            if(tmp<=n) nums[tmp-1]=-abs(nums[tmp-1]); // -1
        }
        for(int i=0;i<n;++i){
            if(nums[i]>0) return i+1;
        }
        return n+1; // 几乎不用
    }
};
```

换位置

```cpp
class Solution {
public:
    int firstMissingPositive(vector<int>& nums) {
        int n=nums.size();
        for(int i=0;i<n;++i){
            while(nums[i]>0 && nums[i]<=n && nums[nums[i]-1]!=nums[i]) // 不是if
                swap(nums[nums[i]-1],nums[i]);
        }
        for(int i=0;i<n;++i){
            if(nums[i]!=i+1) return i+1;
        }
        return n+1; // 几乎不用
    }
};
```
