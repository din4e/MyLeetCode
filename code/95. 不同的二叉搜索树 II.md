class Solution {
public:
    vector<TreeNode*> dfs(int l,int r){
        if(l>r)
            return {nullptr};
        vector<TreeNode*> res;
        for(int i=l;i<=r;++i){
            auto left=dfs(l,i-1); // !
            auto right=dfs(i+1,r); // !
            for(auto &itl:left){
                for(auto &itr:right){
                    TreeNode *root=new TreeNode(i);
                    root->left=itl;
                    root->right=itr;
                    res.emplace_back(root);
                }
            }
        }
        return res; // å§æ§½è¿”å› vector<*TreeNode>
    }
    vector<TreeNode*> generateTrees(int n) {
        if(n==0) return {}; // ä¸æ˜¯ {nullptr}
        return dfs(1,n);
    }
};
