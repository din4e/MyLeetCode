# [46. 全排列](https://leetcode-cn.com/problems/permutations/)

```cpp
class Solution {
public:
    void dfs(vector<vector<int>> &res,int i,int j,vector<int> nums){
        if(i==nums.size())
            return ;
        if(i==nums.size()-1){
            res.push_back(nums);
            return ;
        }
        swap(nums[i],nums[j]);
        for(int k=i+1;k<nums.size();++k)
            dfs(res,i+1,k,nums);
        return ;
    }
    vector<vector<int>> permute(vector<int>& nums) {
        vector<vector<int>> res;
        for(int i=0;i<nums.size();++i)
            dfs(res,0,i,nums);
        return res;
    }
};
```
