# [208. 实现 Trie (前缀树)](https://leetcode-cn.com/problems/implement-trie-prefix-tree/submissions/)

```cpp
class Trie {
private:
    bool isEnd;
    Trie *next[26]; // 需要初始化
public:
    /** Initialize your data structure here. */
    Trie() {
        isEnd = false;
        memset(next, 0, sizeof(next)); // important
    }

    /** Inserts a word into the trie. */
    void insert(string word) {
        Trie *node=this;
        for(auto &c:word){
            if(node->next[c-'a']==nullptr)
                node->next[c-'a']=new Trie();
            node=node->next[c-'a']; // !
        }
        node->isEnd=true;
    }

    /** Returns if the word is in the trie. */
    bool search(string word) {
        Trie *node=this;
        for(auto &c:word){
            node=node->next[c-'a']; // !
            if(node==nullptr)
                return false;
        }
        return node->isEnd;
    }

    /** Returns if there is any word in the trie that starts with the given prefix. */
    bool startsWith(string prefix) {
        Trie *node=this;
        for(auto &c:prefix){
            node=node->next[c-'a']; // !
            if(node==nullptr)
                return false;
        }
        return true;
    }
};

 * obj->insert(word);
 * bool param_2 = obj->search(word);
 * bool param_3 = obj->startsWith(prefix);
 */
```
