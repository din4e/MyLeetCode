# [97. 交错字符串](https://leetcode-cn.com/problems/interleaving-string/submissions/)

+ 思路介绍：字符串匹配 DP

```cpp
class Solution {
public:
    bool isInterleave(string s1, string s2, string s3) {
        int l1=s1.size(),l2=s2.size(),l3=s3.size();
        if(l1+l2!=l3)  return false;
        vector<vector<bool>> dp(l1+1,vector<bool>(l2+1,false));
        dp[0][0]=true;
        for(int i=0;i<l1;++i)
            if(s1[i]==s3[i]) dp[i+1][0]=true;
            else break;
        for(int j=0;j<l2;++j)
            if(s2[j]==s3[j]) dp[0][j+1]=true;
            else break;
        for(int i=0;i<l1;++i){
            for(int j=0;j<l2;++j){
                if(s1[i]==s3[i+j+1]&&dp[i][j+1]) dp[i+1][j+1]=true;
                if(s2[j]==s3[i+j+1]&&dp[i+1][j]) dp[i+1][j+1]=true;
            }
        }
        return dp[l1][l2];
    }
};
```
