#include<bits/stdc++.h>
using namespace std;
void split(string &s,vector<vector<int>> &e){
    string::size_type p1,p2,p3;
    int a,b;
    p1=2;p2=s.find(',',p1);p3=s.find(']',p2);
    while(string::npos!=p2&&string::npos!=p3){
        //cout<<p1<<','<<p2<<','<<p3<<'|';
        a=stoi(s.substr(p1,p2-p1))-1;++p2;
        b=stoi(s.substr(p2,p3-p2))-1;p1=p3+3;
        //cout<<a<<" "<<b<<"|";
        e[a].push_back(b);
        e[b].push_back(a);
        p2=s.find(',',p1);
        p3=s.find(']',p2);
    }
}
int main(){
    int n;
    cin>>n;getchar();
    string s;
    getline(cin,s);
    vector<vector<int>> e(n);
    split(s,e);
    //for(auto it:e) {for(auto it2:it) cout<<it2<<" ";cout<<'\n';}
    vector<int> isVisit(n,0);
    for(int i=0;i<n;i++){
        if(isVisit[i]==0){
            queue<int> q;
            isVisit[i]=1;
            q.push(i);
            while(!q.empty()){
                int tmp=q.front();
                q.pop();
                for(auto it:e[tmp]){
                    if(isVisit[it]==0){
                        isVisit[it]=-1*isVisit[tmp];
                        q.push(it);
                    }
                    else if(isVisit[it]==isVisit[tmp]){
                            cout<<"false";
                            return 0;
                        }
                }
            }
        }
    }
    cout<<"true";
    return 0;
}

#include<bits/stdc++.h>
using namespace std;
void split(string &s,vector<int> &v){
    string::size_type p1,p2;
    p2=s.find(',');p1=1;
    while(string::npos!=p2&&p2<s.size()){
        v.emplace_back(stoi(s.substr(p1,p2-p1)));
        p1=p2+2;p2=s.find(',',p1);
    }
}
int main(){
    string ss;
    vector<int> v;
    
    getline(cin,ss);
    ss[ss.size()-1]=',';
    split(ss,v);
    
    int n=v.size();
    vector<int> res(n,0);
    vector<int> s;
    s.push_back(0);
    for(int i=1;i<n;++i){
        while(!s.empty()&&v[s.back()]<v[i]){
            res[s.back()]=i-s.back();
            s.pop_back();
        }
        s.push_back(i);
    }
    cout<<'['<<res[0];
    for(int i=1;i<res.size();++i) cout<<", "<<res[i];
    cout<<']';
    return 0;
}

#include<bits/stdc++.h>
using namespace std;
void split(string &s,vector<int> &v){
    string::size_type p1,p2;
    p2=s.find(',');p1=0;
    while(string::npos!=p2){
        v.push_back(stoi(s.substr(p1+1,p2-p1-1)));
        cout<<v.back();
        p1=p2+1;p2=s.find(',',p1);
    }
}
int main(){
    int n;
    cin>>n;
    string res[n+1];
    res[1]="1";
    int cnt=0;
    for(int k=2;k<=n;++k){
        res[k]="";
        cnt=1;
        for(int i=0;i<res[k-1].size();++i){
            if(res[k-1][i]==res[k-1][i+1]) ++cnt;
            else{
                res[k]+=to_string(cnt)+res[k-1][i];cnt=1;
            }
        }
    }
    cout<<res[n];
    return 0;
}
